TagRegistry:
  TagRegistry.register(key, tag)           % hard-errors on duplicate key
  TagRegistry.get(key) -> Tag
  TagRegistry.findByKind('sensor'|'state'|'monitor'|'composite') -> cell
  TagRegistry.clear()                      % used in test + teardown

SensorTag:
  SensorTag(key, 'Name', name, 'Units', units, 'RawSource', struct(...))   % X/Y via updateData or pipeline
  sensor.updateData(x, y)

StateTag:
  StateTag(key, 'Name', name, 'Labels', cellstr, 'RawSource', struct(...))

MonitorTag:
  MonitorTag(key, parentTag, conditionFn, ...
             'MinDuration', secs, 'AlarmOn', v, 'AlarmOff', v, ...
             'EventStore', store, 'Criticality', 'warning'|'critical')
  % conditionFn: @(y) logical — fires events on 0->1 rising edges via EventStore

CompositeTag:
  CompositeTag(key, 'AggregateMode', 'and'|'or'|'majority'|'worst'|'severity', 'Name', ...)
  composite.addChild(monitorOrKey, 'Weight', w)   % cycle-detected at addChild time

LiveTagPipeline (Phase 1012 D-07/D-12/D-13/D-14):
  p = LiveTagPipeline(rawDir, tagsDir, ...     % tagsDir receives per-tag .mat
                      'TickRate', 1.0, ...
                      'Registry', TagRegistry.instance())
  p.start()
  p.stop()
  % tick: scans rawDir for modified .dat files -> parses incremental rows -> routes via RawSource binding on each Tag

EventStore:
  EventStore()
  store.append(event)
  store.eventsForTag(tagKey) -> Event[]
