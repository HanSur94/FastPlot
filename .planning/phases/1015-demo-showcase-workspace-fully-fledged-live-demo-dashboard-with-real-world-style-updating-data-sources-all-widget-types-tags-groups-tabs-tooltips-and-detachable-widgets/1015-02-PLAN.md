    ```
    function engine = buildDashboard(ctx)
        engine = DashboardEngine('Theme', DashboardTheme.dark, 'Live', true, 'LiveInterval', 1.0);
        engine.addPage('Overview');
        engine.addPage('Feed Line');
        engine.addPage('Reactor');
        engine.addPage('Cooling');
        engine.addPage('Events');
        engine.addPage('Diagnostics');

        engine.switchPage(1); buildOverviewPage(engine, ctx);
        engine.switchPage(2); buildFeedLinePage(engine, ctx);
        engine.switchPage(3); buildReactorPage(engine, ctx);
        engine.switchPage(4); buildCoolingPage(engine, ctx);
        engine.switchPage(5); buildEventsPage(engine, ctx);
        engine.switchPage(6); buildDiagnosticsPage(engine, ctx);

        engine.switchPage(1);
        fig = figure('Name', 'FastSense Industrial Plant Demo', 'Position', [100 100 1600 950]);
        engine.render(fig);

        % D-09 pre-detach main plot (best-effort, Octave-safe)
        try
            mainPlot = firstWidgetByTag_(engine, 'reactor.pressure');
            if ~isempty(mainPlot), engine.detachWidget(mainPlot); end
        catch
        end

        engine.start();
        ctx.engine = engine;
        set(fig, 'CloseRequestFcn', @(src,~) demoClose_(src, ctx));
    end

    function w = firstWidgetByTag_(engine, key)
        %FIRSTWIDGETBYTAG_ Search all pages for a widget whose Tag matches `key`.
        %   Uses engine.allPageWidgets() (PUBLIC method, line 1112 of DashboardEngine.m)
        %   which returns a cell of all widgets across all pages — safer than
        %   iterating engine.Pages directly and works in both single- and multi-page mode.
        w = [];
        ws = engine.allPageWidgets();
        for i = 1:numel(ws)
            wi = ws{i};
            if isprop(wi, 'Tag') && ischar(wi.Tag) && strcmp(wi.Tag, key)
                w = wi; return;
            end
        end
    end

    function demoClose_(fig, ctx)
        try teardownDemo(ctx); catch; end
        try delete(fig); catch; end
    end
    ```

    NOTE: `engine.allPageWidgets()` is a verified PUBLIC method on DashboardEngine (confirmed at line 1112 of libs/Dashboard/DashboardEngine.m). It concatenates `obj.Pages{i}.Widgets` across all pages. Using it avoids re-implementing page iteration and keeps this helper robust in single-page mode too.
