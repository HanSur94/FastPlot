    ```matlab
    function test_demo_industrial_plant_smoke()
        fprintf('test_demo_industrial_plant_smoke:\n');
        oldVis = get(0, 'defaultfigurevisible');
        set(0, 'defaultfigurevisible', 'off');
        beforeTimers = numel(timerfindall());
        ctx = [];
        try
            % Test 1: boot
            ctx = run_demo();
            assert(~isempty(ctx.engine), 'engine should be populated');
            assert(~isempty(ctx.pipeline), 'pipeline should be populated');

            % Test 2: widgets rendered (use verified public allPageWidgets() method)
            ws = ctx.engine.allPageWidgets();
            nW = numel(ws);
            assert(nW >= 25, sprintf('expected >=25 widgets, got %d', nW));

            % Test 3: event fires
            fid = fopen(fullfile(ctx.rawDir, 'reactor.pressure.dat'), 'a');
            for k=1:10, fprintf(fid, '%.3f,%.6f\n', now*86400 + k*0.1, 19.5); end
            fclose(fid);
            pause(4);
            evs = ctx.store.eventsForTag('reactor.pressure.critical');
            assert(numel(evs) >= 1, 'no reactor.pressure.critical events fired');

            % Test 4: composite resolves
            h = TagRegistry.get(ctx.plantHealthKey);
            v = h.valueAt(now*86400);
            assert(isnumeric(v) && isscalar(v) && ~isnan(v), 'plant.health unresolved');

            fprintf('    All 4 sub-tests passed.\n');
        catch ERR
            try teardownDemo(ctx); catch; end
            set(0, 'defaultfigurevisible', oldVis);
            rethrow(ERR);
        end

        % Test 5: teardown leaves no dangling timers
        teardownDemo(ctx);
        set(0, 'defaultfigurevisible', oldVis);
        afterTimers = numel(timerfindall());
        assert(afterTimers == beforeTimers, ...
            sprintf('dangling timers: before=%d after=%d', beforeTimers, afterTimers));
        fprintf('    Teardown clean.\n');
    end
    ```
